آموزش جامع دستورات جاوا اسکریپت

این صفحه شامل مهم‌ترین دستورات و مفاهیم جاوا اسکریپت به همراه مثال‌های عملی است.

۱. انتخاب عناصر DOM

getElementById()

// انتخاب عنصر با ID خاص
const titleElement = document.getElementById('title');
titleElement.style.color = '#63b3ed';
titleElement.textContent = 'عنوان تغییر کرد!';

querySelector() و querySelectorAll()

// انتخاب اولین عنصر منطبق با selector
const firstItem = document.querySelector('.item');

// انتخاب تمام عناصر منطبق با selector
const allItems = document.querySelectorAll('.item');

۲. متغیرها و ثابت‌ها

// تعریف متغیر با var (قدیمی)
var oldVariable = 'من با var تعریف شده‌ام';

// تعریف متغیر با let (جدید - قابل تغییر)
let counter = 0;
counter = 1; // قابل تغییر است

// تعریف ثابت با const (غیرقابل تغییر)
const PI = 3.14159;
// PI = 3; // خطا - ثابت قابل تغییر نیست

۳. انواع داده‌ها

// رشته (String)
let name = "جان";

// عدد (Number)
let age = 25;

// بولین (Boolean)
let isStudent = true;

// آرایه (Array)
let fruits = ["سیب", "موز", "پرتقال"];

// شیء (Object)
let person = {
    firstName: "جان",
    lastName: "دو",
    age: 30
};

// تهی (Null)
let emptyValue = null;

// تعریف نشده (Undefined)
let notDefined;

۴. عملگرها

// عملگرهای محاسباتی
let sum = 5 + 3;        // 8
let difference = 10 - 4; // 6
let product = 2 * 6;     // 12
let quotient = 15 / 3;   // 5
let remainder = 10 % 3;  // 1

// عملگرهای مقایسه‌ای
let isEqual = 5 == "5";      // true (برابری مقدار)
let isStrictEqual = 5 === "5"; // false (برابری مقدار و نوع)
let isGreater = 10 > 5;      // true
let isLessOrEqual = 5 <= 5;  // true

// عملگرهای منطقی
let andResult = true && false; // false
let orResult = true || false;  // true
let notResult = !true;         // false

۵. شرط‌ها

if...else

let score = 85;

if (score >= 90) {
    console.log("عالی!");
} else if (score >= 70) {
    console.log("خوب");
} else {
    console.log("نیاز به بهبود");
}

switch

let day = 3;
let dayName;

switch(day) {
    case 1:
        dayName = "شنبه";
        break;
    case 2:
        dayName = "یکشنبه";
        break;
    case 3:
        dayName = "دوشنبه";
        break;
    default:
        dayName = "روز نامشخص";
}

۶. حلقه‌ها

for loop

for (let i = 0; i < 5; i++) {
    console.log(i);
}

while loop

let count = 0;
while (count < 5) {
    console.log(count);
    count++;
}

for...of loop (برای آرایه‌ها)

let fruits = ["سیب", "موز", "پرتقال"];
for (let fruit of fruits) {
    console.log(fruit);
}

۷. توابع

تعریف تابع

// تعریف معمولی تابع
function greet(name) {
    return "سلام " + name + "!";
}

// تابع بی‌نام (anonymous function)
const multiply = function(a, b) {
    return a * b;
};

// تابع arrow (جدید)
const divide = (a, b) => a / b;

// فراخوانی توابع
console.log(greet("جان")); // سلام جان!
console.log(multiply(4, 5)); // 20
console.log(divide(10, 2)); // 5

۸. رویدادها (Events)

// اضافه کردن event listener
const button = document.getElementById('myButton');

button.addEventListener('click', function() {
    alert('دکمه کلیک شد!');
});

// رویدادهای رایج
element.onclick = function() {};    // کلیک
element.onmouseover = function() {}; // هاور ماوس
element.onkeydown = function() {};  // فشردن کلید
element.onload = function() {};     // بارگذاری صفحه

نکته مهم:

جاوا اسکریپت یک زبان برنامه‌نویسی پویا و انعطاف‌پذیر است که هم در سمت کلاینت (مرورگر) و هم در سمت سرور (Node.js) قابل استفاده است.
